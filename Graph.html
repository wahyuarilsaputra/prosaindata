Materi : Graph#

import networkx as nx
import pandas as pd
import matplotlib.pyplot as plt

Membuat Graph#

Menggunakan network x

G = nx.Graph()
G.add_edges_from([(1,2),(1,3),(1,4),(2,3),(3,4),(4,5),(4,6),(5,6),(5,7),(5,8),(6,7),(6,8),(7,8),(7,9)])

Labeling Graph#

nx.draw(G, with_labels=True)
plt.show()
